import requests
import json

# Replace this with your actual API key when you get it
API_KEY = "YOUR_API_KEY"
BASE_URL = "https://api.canlii.org/v1"

def list_legislation_databases():
    """List all legislation databases (federal, provincial, etc.)"""
    url = f"{BASE_URL}/legislationBrowse/en/?api_key={API_KEY}"
    response = requests.get(url)
    if response.status_code == 200:
        data = response.json()
        print("=== Legislation Databases ===")
        for db in data.get("legislationDatabases", [])[:10]:
            print(f"  {db['databaseId']} - {db['name']} ({db['jurisdiction']})")
    else:
        print(f"Error: {response.status_code} - {response.text}")

def search_legislation(database_id, query):
    """Search for legislation within a specific database"""
    url = f"{BASE_URL}/legislationBrowse/en/{database_id}/?api_key={API_KEY}"
    response = requests.get(url)
    if response.status_code == 200:
        data = response.json()
        print(f"\n=== Legislation in {database_id} ===")
        for leg in data.get("legislations", [])[:10]:
            print(f"  {leg['legislationId']} - {leg['title']}")
    else:
        print(f"Error: {response.status_code} - {response.text}")

def get_case_citations(database_id, case_id):
    """Get cases and legislation cited by a specific case"""
    url = f"{BASE_URL}/caseCitator/en/{database_id}/{case_id}/citedLegislation?api_key={API_KEY}"
    response = requests.get(url)
    if response.status_code == 200:
        data = response.json()
        print(f"\n=== Legislation Cited by Case ===")
        for leg in data.get("citedLegislations", [])[:10]:
            print(f"  {leg['title']} - {leg['citation']}")
    else:
        print(f"Error: {response.status_code} - {response.text}")
